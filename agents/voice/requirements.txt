# LiveKit Voice Agent Worker — Separate Dependencies
#
# The voice agent worker runs as a SEPARATE process from the FastAPI server.
# It has its own opentelemetry-sdk version that conflicts with crewai.
#
# The worker does NOT import crewai — it calls the FastAPI /api/v1/rag/query
# endpoint over HTTP instead, keeping a clean dependency boundary.
#
# Setup:
#   python -m venv .venv-voice
#   .venv-voice/Scripts/activate     (Windows)
#   source .venv-voice/bin/activate  (Linux/Mac)
#   pip install -r agents/voice/requirements.txt
#
# Run:
#   python -m agents.voice.worker dev

# LiveKit Agents SDK 1.4+ with plugins
livekit-agents[silero,turn-detector]~=1.4
livekit-plugins-noise-cancellation~=0.2

# HTTP client (calls FastAPI RAG endpoint instead of importing CrewAI)
httpx>=0.27.0

# Minimal shared deps
pydantic>=2.9.0
python-dotenv>=1.0.0
